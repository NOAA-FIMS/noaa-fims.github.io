428 Type CalculateSBPR0(
429 std::shared_ptr<fims_popdy::Population<Type>> &population) {
430 std::map<std::string, fims::Vector<Type>> &dq_ =
431 this->GetPopulationDerivedQuantities(population->GetId());
432
433 std::vector<Type> numbers_spr(population->n_ages, 1.0);
434 Type phi_0 = 0.0;
435 phi_0 += numbers_spr[0] * population->proportion_female[0] *
436 dq_["proportion_mature_at_age"][0] *
437 population->growth->evaluate(population->ages[0]);
438 for (size_t a = 1; a < (population->n_ages - 1); a++) {
439 numbers_spr[a] = numbers_spr[a - 1] * fims_math::exp(-population->M[a]);
440 phi_0 += numbers_spr[a] * population->proportion_female[a] *
441 dq_["proportion_mature_at_age"][a] *
442 population->growth->evaluate(population->ages[a]);
443 }
444
445 numbers_spr[population->n_ages - 1] =
446 (numbers_spr[population->n_ages - 2] *
447 fims_math::exp(-population->M[population->n_ages - 2])) /
448 (1 - fims_math::exp(-population->M[population->n_ages - 1]));
449 phi_0 +=
450 numbers_spr[population->n_ages - 1] *
451 population->proportion_female[population->n_ages - 1] *
452 dq_["proportion_mature_at_age"][population->n_ages - 1] *
453 population->growth->evaluate(population->ages[population->n_ages - 1]);
454
455 return phi_0;
456 }

462 void CalculateRecruitment(
463 std::shared_ptr<fims_popdy::Population<Type>> &population,
464 size_t i_age_year, size_t year, size_t i_dev) {
465 std::map<std::string, fims::Vector<Type>> &dq_ =
466 this->GetPopulationDerivedQuantities(population->GetId());
467
468 Type phi0 = CalculateSBPR0(population);
469
470 if (i_dev == population->n_years) {
471 dq_["numbers_at_age"][i_age_year] =
472 population->recruitment->evaluate_mean(
473 dq_["spawning_biomass"][year - 1], phi0);
474 /*the final year of the time series has no data to inform recruitment
475 devs, so this value is set to the mean recruitment.*/
476 } else {
477 // Why are we using evaluate_mean, how come a virtual function was
478 // changed? AMH: there are now two virtual functions: evaluate_mean and
479 // evaluate_process (see below)
480 population->recruitment->log_expected_recruitment[year - 1] =
481 fims_math::log(population->recruitment->evaluate_mean(
482 dq_["spawning_biomass"][year - 1], phi0));
483
484 dq_["numbers_at_age"][i_age_year] = fims_math::exp(
485 population->recruitment->process->evaluate_process(year - 1));
486 }
487
488 dq_["expected_recruitment"][year] = dq_["numbers_at_age"][i_age_year];

664 void evaluate_age_comp() {
665 fleet_iterator fit;
666 for (fit = this->fleets.begin(); fit != this->fleets.end(); ++fit) {
667 std::map<std::string, fims::Vector<Type>> &fdq_ =
668 this->GetFleetDerivedQuantities((*fit).second->GetId());
669
670 std::shared_ptr<fims_popdy::Fleet<Type>> &fleet = (*fit).second;
671 for (size_t y = 0; y < fleet->n_years; y++) {
672 Type sum = static_cast<Type>(0.0);
673 Type sum_obs = static_cast<Type>(0.0);
674 // robust_add is a small value to add to expected composition
675 // proportions at age to stabilize likelihood calculations
676 // when the expected proportions are close to zero.
677 // Type robust_add = static_cast<Type>(0.0); // zeroed out before
678 // testing 0.0001; sum robust is used to calculate the total sum of
679 // robust additions to ensure that proportions sum to 1. Type robust_sum
680 // = static_cast<Type>(1.0);
681
682 for (size_t a = 0; a < fleet->n_ages; a++) {
683 size_t i_age_year = y * fleet->n_ages + a;
684 // Here we have a check to determine if the age comp
685 // should be calculated from the retained landings or
686 // the total population. These values are slightly different.
687 // In the future this will have more impact as we implement
688 // timing rather than everything occurring at the start of
689 // the year.
690 if (fleet->fleet_observed_landings_data_id_m == -999) {
691 fdq_["agecomp_expected"][i_age_year] =
692 fdq_["index_numbers_at_age"][i_age_year];
693 } else {
694 fdq_["agecomp_expected"][i_age_year] =
695 fdq_["landings_numbers_at_age"][i_age_year];
696 }
697 sum += fdq_["agecomp_expected"][i_age_year];
698 // robust_sum -= robust_add;
699
700 // This sums over the observed age composition data so that
701 // the expected age composition can be rescaled to match the
702 // total number observed. The check for na values should not
703 // be needed as individual years should not have missing data.
704 // This is need to be re-explored if/when we modify FIMS to
705 // allow for composition bins that do not match the population
706 // bins.
707 if (fleet->fleet_observed_agecomp_data_id_m != -999) {
708 if (fleet->observed_agecomp_data->at(i_age_year) !=
709 fleet->observed_agecomp_data->na_value) {
710 sum_obs += fleet->observed_agecomp_data->at(i_age_year);
711 }
712 }
713 }
714 for (size_t a = 0; a < fleet->n_ages; a++) {
715 size_t i_age_year = y * fleet->n_ages + a;
716 fdq_["agecomp_proportion"][i_age_year] =
717 fdq_["agecomp_expected"][i_age_year] / sum;
718 // robust_add + robust_sum * this->agecomp_expected[i_age_year] / sum;
719
720 if (fleet->fleet_observed_agecomp_data_id_m != -999) {
721 fdq_["agecomp_expected"][i_age_year] =
722 fdq_["agecomp_proportion"][i_age_year] * sum_obs;
723 }
724 }
725 }

837 virtual void Evaluate() {
838 /*
839 Sets derived vectors to zero
840 Performs parameters transformations
841 Sets recruitment deviations to mean 0.
842 */
843 Prepare();
844 /*
845 start at year=0, age=0;
846 here year 0 is the estimated initial population structure and age 0 are
847 recruits loops start at zero with if statements inside to specify unique
848 code for initial structure and recruitment 0 loops. Could also have started
849 loops at 1 with initial structure and recruitment setup outside the loops.
850
851 year loop is extended to <= n_years because SSB is calculated as the start
852 of the year value and by extending one extra year we get estimates of the
853 population structure at the end of the final year. An alternative approach
854 would be to keep initial numbers at age in it's own vector and each year to
855 include the population structure at the end of the year. This is likely a
856 null point given that we are planning to modify to an event/stanza based
857 structure in later milestones which will eliminate this confusion by
858 explicitly referencing the exact date (or period of averaging) at which any
859 calculation or output is being made.
860 */
861 for (size_t p = 0; p < this->populations.size(); p++) {
862 std::shared_ptr<fims_popdy::Population<Type>> &population =
863 this->populations[p];
864 std::map<std::string, fims::Vector<Type>> &pdq_ =
865 this->GetPopulationDerivedQuantities(population->GetId());
866 // CAAPopulationProxy<Type>& population = this->populations_proxies[p];
867
868 for (size_t y = 0; y <= population->n_years; y++) {
869 for (size_t a = 0; a < population->n_ages; a++) {
870 /*
871 index naming defines the dimensional folding structure
872 i.e. i_age_year is referencing folding over years and ages.
873 */
874 size_t i_age_year = y * population->n_ages + a;
875 /*
876 Mortality rates are not estimated in the final year which is
877 used to show expected population structure at the end of the model
878 period. This is because biomass in year i represents biomass at the
879 start of the year. Should we add complexity to track more values such
880 as start, mid, and end biomass in all years where, start biomass=end
881 biomass of the previous year? Referenced above, this is probably not
882 worth exploring as later milestone changes will eliminate this
883 confusion.
884 */
885 if (y < population->n_years) {
886 /*
887 First thing we need is total mortality aggregated across all fleets
888 to inform the subsequent catch and change in numbers at age
889 calculations. This is only calculated for years < n_years as these
890 are the model estimated years with data. The year loop extends to
891 y=n_years so that population numbers at age and SSB can be
892 calculated at the end of the last year of the model
893 */
894 CalculateMortality(population, i_age_year, y, a);
895 }
896 CalculateMaturityAA(population, i_age_year, a);
897 /* if statements needed because some quantities are only needed
898 for the first year and/or age, so these steps are included here.
899 */
900 if (y == 0) {
901 // Initial numbers at age is a user input or estimated parameter
902 // vector.
903 CalculateInitialNumbersAA(population, i_age_year, a);
904
905 if (a == 0) {
906 pdq_["unfished_numbers_at_age"][i_age_year] =
907 fims_math::exp(population->recruitment->log_rzero[0]);
908 } else {
909 CalculateUnfishedNumbersAA(population, i_age_year, a - 1, a);
910 }
911
912 /*
913 Fished and unfished biomass vectors are summing biomass at
914 age across ages.
915 */
916
917 CalculateBiomass(population, i_age_year, y, a);
918
919 CalculateUnfishedBiomass(population, i_age_year, y, a);
920
921 /*
922 Fished and unfished spawning biomass vectors are summing biomass at
923 age across ages to allow calculation of recruitment in the next
924 year.
925 */
926
927 CalculateSpawningBiomass(population, i_age_year, y, a);
928
929 CalculateUnfishedSpawningBiomass(population, i_age_year, y, a);
930
931 /*
932 Expected recruitment in year 0 is numbers at age 0 in year 0.
933 */
934
935 pdq_["expected_recruitment"][i_age_year] =
936 pdq_["numbers_at_age"][i_age_year];
937 } else {
938 if (a == 0) {
939 // Set the nrecruits for age a=0 year y (use pointers instead of
940 // functional returns) assuming fecundity = 1 and 50:50 sex ratio
941 CalculateRecruitment(population, i_age_year, y, y);
942 pdq_["unfished_numbers_at_age"][i_age_year] =
943 fims_math::exp(population->recruitment->log_rzero[0]);
944 } else {
945 size_t i_agem1_yearm1 = (y - 1) * population->n_ages + (a - 1);
946 CalculateNumbersAA(population, i_age_year, i_agem1_yearm1, a);
947 CalculateUnfishedNumbersAA(population, i_age_year, i_agem1_yearm1,
948 a);
949 }
950 CalculateBiomass(population, i_age_year, y, a);
951 CalculateSpawningBiomass(population, i_age_year, y, a);
952
953 CalculateUnfishedBiomass(population, i_age_year, y, a);
954 CalculateUnfishedSpawningBiomass(population, i_age_year, y, a);
955 }
956
957 /*
958 Here composition, total catch, and index values are calculated for all
959 years with reference data. They are not calculated for y=n_years as
960 there is this is just to get final population structure at the end of
961 the terminal year.
962 */
963 if (y < population->n_years) {
964 CalculateLandingsNumbersAA(population, i_age_year, y, a);
965 CalculateLandingsWeightAA(population, y, a);
966 CalculateLandings(population, y, a);
967
968 CalculateIndexNumbersAA(population, i_age_year, y, a);
969 CalculateIndexWeightAA(population, y, a);
970 CalculateIndex(population, i_age_year, y, a);
971 }
972 }
973 }
974 }
975 evaluate_age_comp();
976 evaluate_length_comp();
977 evaluate_index();
978 evaluate_landings();
