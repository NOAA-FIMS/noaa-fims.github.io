339 } else {
340 this->expected_mean[i].final_value_m = dnorm->expected_mean[i];
341 }
342 }
343
344 this->lpdf_vec = RealVector(dnorm->report_lpdf_vec.size());
345 if (this->expected_values.size() == 1) {
346 this->expected_values.resize(dnorm->expected_values.size());

348 if (this->x.size() == 1) {
349 size_t nx = dnorm->get_n_x();
350 this->x.resize(nx);
351 }
352
353 for (R_xlen_t i = 0; i < this->lpdf_vec.size(); i++) {
354 this->lpdf_vec[i] = dnorm->report_lpdf_vec[i];
355 this->expected_values[i].final_value_m = dnorm->get_expected(i);
356 this->x[i].final_value_m = dnorm->get_observed(i);
357 }

933 }

936
937 std::shared_ptr<fims_info::Information<double>> info =
938 fims_info::Information<double>::GetInstance();
939
940 fims_info::Information<double>::density_components_iterator it;
941
942 // search for density component in Information

946 FIMS_WARNING_LOG("DmultinomDistributions " + fims::to_string(this->id_m) +

962 this->x.resize(nx);
963 }

965 this->lpdf_vec[i] = dmultinom->report_lpdf_vec[i];

967 if (dmultinom->input_type != "data") {
968 this->x[i].final_value_m = dmultinom->get_observed(i);
