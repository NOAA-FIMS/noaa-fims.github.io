243 if (this->finalized) {
244 // log warning that finalize has been called more than once.
245 FIMS_WARNING_LOG("Population " + fims::to_string(this->id) +
246 " has been finalized already.");
247 }
248
249 this->finalized = true; // indicate this has been called already
250
251 std::shared_ptr<fims_info::Information<double>> info =
252 fims_info::Information<double>::GetInstance();
253
254 fims_info::Information<double>::population_iterator it;
255
256 it = info->populations.find(this->id);
257
258 std::shared_ptr<fims_popdy::Population<double>> pop =
259 info->populations[this->id];
260 it = info->populations.find(this->id);
261 if (it == info->populations.end()) {
262 FIMS_WARNING_LOG("Population " + fims::to_string(this->id) +
263 " not found in Information.");
264 return;
265 } else {
266 for (size_t i = 0; i < this->log_M.size(); i++) {
267 if (this->log_M[i].estimation_type_m.get() == "constant") {
268 this->log_M[i].final_value_m = this->log_M[i].initial_value_m;
269 } else {
270 this->log_M[i].final_value_m = pop->log_M[i];
271 }
272 }
273
274 for (size_t i = 0; i < this->log_f_multiplier.size(); i++) {
275 if (this->log_f_multiplier[i].estimation_type_m.get() == "constant") {
276 this->log_f_multiplier[i].final_value_m =
277 this->log_f_multiplier[i].initial_value_m;
278 } else {
279 this->log_f_multiplier[i].final_value_m = pop->log_f_multiplier[i];
