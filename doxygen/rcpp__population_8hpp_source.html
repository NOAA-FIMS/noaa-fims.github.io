315 std::stringstream ss;
316
317 // set relative info
318 population->id = this->id;
319 population->n_years = this->n_years.get();
320 population->n_fleets = this->n_fleets.get();
321 // only define ages if n_ages greater than 0
322 if (this->n_ages.get() > 0) {
323 population->n_ages = this->n_ages.get();
324 if (this->n_ages.get() == this->ages.size()) {
325 population->ages.resize(this->n_ages.get());
326 } else {
327 warning("The ages vector is not of size n_ages.");
328 }
329 }
330
331 fleet_ids_iterator it;
332 for (it = this->fleet_ids->begin(); it != this->fleet_ids->end(); it++) {
333 population->fleet_ids.insert(*it);
334 }
335
336 population->growth_id = this->growth_id.get();
337 population->recruitment_id = this->recruitment_id.get();
338 population->maturity_id = this->maturity_id.get();
339 population->log_M.resize(this->log_M.size());
340 population->log_init_naa.resize(this->log_init_naa.size());
341 for (size_t i = 0; i < log_M.size(); i++) {
342 population->log_M[i] = this->log_M[i].initial_value_m;
343 if (this->log_M[i].estimation_type_m.get() == "fixed_effects") {
344 ss.str("");
345 ss << "Population." << this->id << ".log_M." << this->log_M[i].id_m;
346 info->RegisterParameterName(ss.str());
347 info->RegisterParameter(population->log_M[i]);
348 }
349 if (this->log_M[i].estimation_type_m.get() == "random_effects") {
350 ss.str("");
351 ss << "Population." << this->id << ".log_M." << this->log_M[i].id_m;
352 info->RegisterRandomEffectName(ss.str());
353 info->RegisterRandomEffect(population->log_M[i]);
354 }
355 }
356 info->variable_map[this->log_M.id_m] = &(population)->log_M;
357
358 for (size_t i = 0; i < log_init_naa.size(); i++) {
359 population->log_init_naa[i] = this->log_init_naa[i].initial_value_m;
360 if (this->log_init_naa[i].estimation_type_m.get() == "fixed_effects") {
361 ss.str("");
362 ss << "Population." << this->id << ".log_init_naa."

364 info->RegisterParameterName(ss.str());
365 info->RegisterParameter(population->log_init_naa[i]);
366 }
367 if (this->log_init_naa[i].estimation_type_m.get() == "random_effects") {
368 ss.str("");

371 info->RegisterRandomEffectName(ss.str());
372 info->RegisterRandomEffect(population->log_init_naa[i]);
373 }
374 }

376
377 for (int i = 0; i < ages.size(); i++) {
378 population->ages[i] = this->ages[i];
379 }
